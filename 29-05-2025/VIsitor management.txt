CREATE DATABASE VisitorManagementDB
USE VisitorManagementDB
CREATE TABLE Visitor(
VisitorId BIGINT PRIMARY KEY NOT NULL,
VisitorName NVARCHAR NOT NULL,
Location INT NOT NULL,
parentCompany NVARCHAR NOT NULL,
Purpose NVARCHAR NOT NULL,
MobileNumber BIGINT NOT NULL,
VisitingDate NVARCHAR NOT NULL,
HostId BIGINT NOT NULL
)


CREATE TABLE Host(
HostId BIGINT PRIMARY KEY NOT NULL,
HostName NVARCHAR NOT NULL,
EmployeeNumber BIGINT NOT NULL,
Department NVARCHAR NOT NULL,
Location NVARCHAR NOT NULL,
ContactNumber BIGINT NOT NULL,
)

CREATE TABLE Location(
LocationId BIGINT PRIMARY KEY NOT NULL,
LocationName NVARCHAR NOT NULL,
)

ALTER TABLE Visitor ADD CONSTRAINT
Visitor_Host_FK FOREIGN KEY(HostId)
REFERENCES Host(HostId)

ALTER TABLE Host ADD CONSTRAINT
Location_Host_FK FOREIGN KEY(Location)
REFERENCES Location(LocationId)

ALTER TABLE Visitor ADD CONSTRAINT
Visitor_Location_FK FOREIGN KEY(Location)
REFERENCES Location(LocationId)
